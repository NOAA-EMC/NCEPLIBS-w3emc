7 c this is the fortran 32 bit version of sbyte.
8 c changes for silicongraphics iris-4d/25
9 c silicongraphics 3.3 fortran 77
10 c march 1991 russell e. jones
11 c national weather service

13  INTEGER IN
14  INTEGER IOUT(*)
15  INTEGER MASKS(32)
16 c
17  SAVE
18 c
19  DATA nbitsw/32/
20 c
21 c DATA masks /z'00000001',z'00000003',z'00000007',z'0000000F',
22 c & z'0000001F',z'0000003F',z'0000007F',z'000000FF',
23 c & z'000001FF',z'000003FF',z'000007FF',z'00000FFF',
24 c & z'00001FFF',z'00003FFF',z'00007FFF',z'0000FFFF',
25 c & z'0001FFFF',z'0003FFFF',z'0007FFFF',z'000FFFFF',
26 c & z'001FFFFF',z'003FFFFF',z'007FFFFF',z'00FFFFFF',
27 c & z'01FFFFFF',z'03FFFFFF',z'07FFFFFF',z'0FFFFFFF',
28 c & z'1FFFFFFF',z'3FFFFFFF',z'7FFFFFFF',z'FFFFFFFF'/
29 c
30 c mask table put in decimal so it will compile on an 32 bit
31 c computer
32 c
33  DATA masks / 1, 3, 7, 15, 31, 63, 127, 255, 511, 1023, 2047,
34  & 4095, 8191, 16383, 32767, 65535, 131071, 262143, 524287,
35  & 1048575, 2097151, 4194303, 8388607, 16777215, 33554431,
36  & 67108863, 134217727, 268435455, 536870911, 1073741823,
37  & 2147483647, -1/
38 c
39 c nbyte must be less than or equal to nbitsw
40 c
41  icon = nbitsw - nbyte
42  IF (icon.LT.0) RETURN
43  mask = masks(nbyte)
44 c
45 c index tells how many words into iout the next byte is to be stored.
46 c
47  index = ishft(iskip,-5)
48 c
49 c ii tells how many bits in from the left side of the word to store it.
50 c
51  ii = mod(iskip,nbitsw)
52 c
53  j = iand(mask,in)
54  movel = icon - ii
55 c
56 c byte is to be stored in middle of word. shift left.
57 c
58  IF (movel.GT.0) THEN
59  msk = ishft(mask,movel)
60  iout(index+1) = ior(iand(not(msk),iout(index+1)),
61  & ishft(j,movel))
62 c
63 c the byte is to be split across a word break.
64 c
65  ELSE IF (movel.LT.0) THEN
66  msk = masks(nbyte+movel)
67  iout(index+1) = ior(iand(not(msk),iout(index+1)),
68  & ishft(j,movel))
69  itemp = iand(masks(nbitsw+movel),iout(index+2))
70  iout(index+2) = ior(itemp,ishft(j,nbitsw+movel))
71 c
72 c byte is to be stored right-adjusted.
73 c
74  ELSE
75  iout(index+1) = ior(iand(not(mask),iout(index+1)),j)
76  ENDIF
77 c
78  RETURN
79  END
