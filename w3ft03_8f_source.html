46 SUBROUTINE w3ft03(FL,HI,STI,STJ,MAXI,MAXJ,ITYPE)
47C
48 REAL FL(MAXI,MAXJ)
49 REAL E (4)
50C
51 SAVE
52C
53 i = sti
54 j = stj
55 di = i
56 dj = j
57 di = sti - di
58 dj = stj - dj
59C
60 hi = 0.
61C TEST FOR POINT OFF GRID.
62 IF (i.LT.1 .OR. i.GT.maxi) GO TO 300
63 IF (j.LT.1 .OR. j.GT.maxj) GO TO 300
64 IF (itype .NE. 2) GO TO 100
65C DO BILINEAR IF POINT IS BETWEEN ULTIMATE AND
66C PENULTIMATE ROWS, WHERE BIQUAD NOT POSSIBLE.
67 IF (i.LT.2 .OR. i.GT.(maxi-1)) GO TO 100
68 IF (j.LT.2 .OR. j.GT.(maxj-1)) GO TO 100
69 GO TO 200
70C
71C BILINEAR.
72 100 CONTINUE
73 hi = fl(i ,j )*(1.-di)*(1.-dj) + fl(i+1,j )*di*(1.-dj)
74 & + fl(i ,j+1)*(1.-di)*dj + fl(i+1,j+1)*di*dj
75 GO TO 300
76C
77 200 CONTINUE
78C BIQUADRATIC.
79 di2 = di*(di-1.)*.25
80 dj2 = dj*(dj-1.)*.25
81 j1 = j - 1
82 DO 250 k=1,4
83 e(k) = fl(i ,j1)*(1.-di-di2) + fl(i+1,j1)*(di-di2)
84 & + (fl(i-1,j1) + fl(i+2,j1))*di2
85 j1 = j1 + 1
86 250 CONTINUE
87 hi = e(2)*(1.-dj-dj2) + e(3)*(dj-dj2) + (e(1) + e(4))*dj2
88C
89 300 CONTINUE
90 RETURN
91 END
