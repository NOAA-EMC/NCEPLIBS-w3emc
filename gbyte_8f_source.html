13 C> SUBROUTINE GBYTE (IPACKD,IUNPKD,NOFF,NBITS)
14 C>
15 C> TO UNPACK A BYTE INTO A TARGET WORD. THE
16 C> UNPACKED BYTE IS RIGHT-JUSTIFIED IN THE
17 C> TARGET WORD, AND THE REMAINDER OF THE
18 C> WORD IS ZERO-FILLED.
19 C>
20 C> @param[in] IPACKD THE WORD OR ARRAY CONTAINING THE BYTE TO BE
21 C> UNPACKED.
22 C>
23 C> @param[out] IUNPKD THE WORD WHICH WILL CONTAIN THE UNPACKED
24 C> BYTE.
25 C>
26 C> @param[in] NOFF THE NUMBER OF BITS TO SKIP, LEFT TO RIGHT,
27 C> IN 'IPACKD' IN ORDER TO LOCATE THE BYTE
28 C> TO BE UNPACKED.
29 C>
30 C> @param[in] NBITS NUMBER OF BITS IN THE BYTE TO BE UNPACKED.
31 C> MAXIMUM OF 64 BITS ON 64 BIT MACHINE, 32
32 C> BITS ON 32 BIT MACHINE.
33 C>
34  SUBROUTINE gbyte(IPACKD,IUNPKD,NOFF,NBITS)
35  INTEGER IPACKD(*)
36  INTEGER IUNPKD
37  INTEGER MASKS(64)
38 C
39  SAVE
40 C
41  DATA ifirst/1/
42  IF(ifirst.EQ.1) THEN
43  CALL w3fi01(lw)
44  nbitsw = 8 * lw
45  jshift = -1 * nint(alog(float(nbitsw)) / alog(2.0))
46  masks(1) = 1
47  DO i=2,nbitsw-1
48  masks(i) = 2 * masks(i-1) + 1
49  ENDDO
50  masks(nbitsw) = -1
51  ifirst = 0
52  ENDIF
53 C
54 C NBITS MUST BE LESS THAN OR EQUAL TO NBITSW
55 C
56  icon = nbitsw - nbits
57  IF (icon.LT.0) RETURN
58  mask = masks(nbits)
59 C
60 C INDEX TELLS HOW MANY WORDS INTO THE ARRAY 'IPACKD' THE NEXT BYTE
61 C APPEARS.
62 C
63  index = ishft(noff,jshift)
64 C
65 C II TELLS HOW MANY BITS THE BYTE IS FROM THE LEFT SIDE OF THE WORD.
66 C
67  ii = mod(noff,nbitsw)
68 C
69 C MOVER SPECIFIES HOW FAR TO THE RIGHT NBITS MUST BE MOVED IN ORDER
70 C
71 C TO BE RIGHT ADJUSTED.
72 C
73  mover = icon - ii
74 C
75  IF (mover.GT.0) THEN
76  iunpkd = iand(ishft(ipackd(index+1),-mover),mask)
77 C
78 C THE BYTE IS SPLIT ACROSS A WORD BREAK.
79 C
80  ELSE IF (mover.LT.0) THEN
81  movel = - mover
82  mover = nbitsw - movel
83  iunpkd = iand(ior(ishft(ipackd(index+1),movel),
84  & ishft(ipackd(index+2),-mover)),mask)
85 C
86 C THE BYTE IS ALREADY RIGHT ADJUSTED.
87 C
88  ELSE
89  iunpkd = iand(ipackd(index+1),mask)
90  ENDIF
91 C
92  RETURN
93  END
